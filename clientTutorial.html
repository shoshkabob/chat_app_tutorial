	{

	      var msghash =
	      {
	        name:name,
	        message:document.getElementById("text").value
	      };
	      var msgstrng = JSON.stringify(msghash);
	      ws.send(msgstrng);
	      textbox.value ="";
	    }
	});
  	var button = document.getElementById('button');
  	button.addEventListener("click",function()
	{

	        message:document.getElementById("text").value
	      };
